Question 2 :- "Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) 

         function original1(string) {
         let array2 = new Map();
         let num = 1;
         for(let count of string) {
            if(array2.has(count)) {
               array2.set(count, array2.get(count) + num);
            }
            else {
               array2.set(count, num);
            }
         }
         for(let [value, number] of array2) {
               console.log(value + " = " + number);
            }
         }
         original1("abcadeecfb")
